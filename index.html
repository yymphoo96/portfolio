wallet_migration.py
class WalletMigration(models.Model):
  _name = 'wallet.account'
  _inherit = 'res.partner'

  def migrate_v8_to_v13(self):
    # Phase 1: schema mapping
    records = self._fetch_legacy()
    for rec in records:
      self._transform(rec)

  def _transform(self, rec):
    # map balance → wallet_balance
    return {
      'wallet_balance': rec.balance,
      'phase': 13,
    }
